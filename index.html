let windValue = 0 let isShieldClosed = false // 初始化 serial.redirectToUSB() wuKong.stopAllMotor() basic.showIcon(IconNames.Happy) // B 鍵手動控制 input.onButtonPressed(Button.B, function () { if (isShieldClosed == false) { basic.showIcon(IconNames.No) wuKong.setMotorSpeed(wuKong.MotorList.M1, 40) basic.pause(4500) wuKong.stopAllMotor() isShieldClosed = true } else { basic.showIcon(IconNames.Yes) wuKong.setMotorSpeed(wuKong.MotorList.M1, -40) basic.pause(4500) wuKong.stopAllMotor() isShieldClosed = false } }) basic.forever(function () { // 直接讀取 P2 原始數值 (0-1023) windValue = pins.analogReadPin(AnalogReadWritePin.P2) // 直接將 P2 數值傳送給 Web App (網頁端會顯示並自動換算) serial.writeValue("wind", windValue) // 自動控制：當 P2數值 > 130 if (windValue > 130 && isShieldClosed == false) { basic.showIcon(IconNames.No) wuKong.setMotorSpeed(wuKong.MotorList.M1, 40) basic.pause(4500) wuKong.stopAllMotor() isShieldClosed = true } basic.pause(500) })
